# 知识库管理平台前端页面设计文档

## 1. 项目概述

### 1.1 设计目标
基于JeecgBoot Vue3框架，参考科亿知识库的页面布局和交互设计，构建现代化、用户友好的企业知识库管理平台前端界面。

### 1.2 设计原则
- **简洁高效**：界面简洁明了，操作流程高效
- **响应式设计**：支持PC、平板、手机多端适配
- **用户体验优先**：注重用户操作习惯和视觉体验
- **组件化开发**：基于Ant Design Vue组件库，保证界面一致性

### 1.3 技术栈
- **前端框架**：Vue 3.5.13 + TypeScript
- **UI组件库**：Ant Design Vue 4.2.6
- **构建工具**：Vite 6.0.7
- **状态管理**：Pinia
- **路由管理**：Vue Router
- **图标库**：Ant Design Icons + 自定义图标

#### 4.1.1 知识卡片组件 (KnowledgeCard)
```vue
<template>
  <div class="knowledge-card" @click="handleCardClick">
    <div class="card-header">
      <div class="title-section">
        <component :is="getTypeIcon(knowledge.type)" class="type-icon" />
        <h3 class="title">{{ knowledge.title }}</h3>
        <a-tag v-if="knowledge.status === 'draft'" color="orange">草稿</a-tag>
        <a-tag v-if="knowledge.isTop" color="red">置顶</a-tag>
      </div>
      <div class="meta">
        <a-avatar :size="24" :src="knowledge.authorAvatar">
          {{ knowledge.author.charAt(0) }}
        </a-avatar>
        <span class="author">{{ knowledge.author }}</span>
        <span class="separator">|</span>
        <span class="category">{{ knowledge.categoryName }}</span>
        <span class="separator">|</span>
        <span class="date">{{ formatDate(knowledge.createTime) }}</span>
      </div>
    </div>

    <div class="card-content">
      <p class="summary">{{ knowledge.summary }}</p>
      <div class="tags" v-if="knowledge.tags?.length">
        <a-tag
          v-for="tag in knowledge.tags.slice(0, 5)"
          :key="tag"
          color="blue"
          @click.stop="handleTagClick(tag)"
        >
          {{ tag }}
        </a-tag>
        <span v-if="knowledge.tags.length > 5" class="more-tags">
          +{{ knowledge.tags.length - 5 }}
        </span>
      </div>
    </div>

    <div class="card-footer">
      <div class="stats">
        <span class="stat-item">
          <EyeOutlined />
          {{ formatNumber(knowledge.viewCount) }}
        </span>
        <span class="stat-item">
          <StarFilled />
          {{ knowledge.rating.toFixed(1) }}
        </span>
        <span class="stat-item">
          <HeartOutlined />
          {{ formatNumber(knowledge.favoriteCount) }}
        </span>
        <span class="stat-item">
          <MessageOutlined />
          {{ formatNumber(knowledge.commentCount) }}
        </span>
      </div>
      <div class="actions">
        <a-button
          type="text"
          :class="{ 'favorited': knowledge.isFavorited }"
          @click.stop="handleFavorite"
        >
          <HeartFilled v-if="knowledge.isFavorited" />
          <HeartOutlined v-else />
        </a-button>
        <a-button type="text" @click.stop="handleShare">
          <ShareAltOutlined />
        </a-button>
        <a-dropdown @click.stop>
          <a-button type="text">
            <MoreOutlined />
          </a-button>
          <template #overlay>
            <a-menu>
              <a-menu-item @click="handleEdit" v-if="knowledge.canEdit">
                <EditOutlined /> 编辑
              </a-menu-item>
              <a-menu-item @click="handleDownload">
                <DownloadOutlined /> 下载
              </a-menu-item>
              <a-menu-item @click="handleReport">
                <FlagOutlined /> 举报
              </a-menu-item>
            </a-menu>
          </template>
        </a-dropdown>
      </div>
    </div>
  </div>
</template>

<script setup lang="ts">
import { defineProps, defineEmits } from 'vue'
import {
  EyeOutlined, StarFilled, HeartOutlined, HeartFilled,
  MessageOutlined, ShareAltOutlined, MoreOutlined,
  EditOutlined, DownloadOutlined, FlagOutlined,
  FileTextOutlined, FileWordOutlined, FilePdfOutlined,
  VideoCameraOutlined, PictureOutlined
} from '@ant-design/icons-vue'

interface Knowledge {
  id: string
  title: string
  author: string
  authorAvatar?: string
  createTime: string
  summary: string
  tags: string[]
  categoryName: string
  type: 'document' | 'word' | 'pdf' | 'video' | 'image'
  status: 'published' | 'draft' | 'reviewing'
  viewCount: number
  rating: number
  favoriteCount: number
  commentCount: number
  isTop: boolean
  isFavorited: boolean
  canEdit: boolean
}

const props = defineProps<{
  knowledge: Knowledge
}>()

const emit = defineEmits<{
  click: [knowledge: Knowledge]
  favorite: [knowledge: Knowledge]
  share: [knowledge: Knowledge]
  tagClick: [tag: string]
  edit: [knowledge: Knowledge]
  download: [knowledge: Knowledge]
  report: [knowledge: Knowledge]
}>()

const getTypeIcon = (type: string) => {
  const iconMap = {
    document: FileTextOutlined,
    word: FileWordOutlined,
    pdf: FilePdfOutlined,
    video: VideoCameraOutlined,
    image: PictureOutlined
  }
  return iconMap[type] || FileTextOutlined
}

const formatNumber = (num: number) => {
  if (num >= 10000) return (num / 10000).toFixed(1) + 'w'
  if (num >= 1000) return (num / 1000).toFixed(1) + 'k'
  return num.toString()
}

const formatDate = (date: string) => {
  const now = new Date()
  const target = new Date(date)
  const diff = now.getTime() - target.getTime()
  const days = Math.floor(diff / (1000 * 60 * 60 * 24))

  if (days === 0) return '今天'
  if (days === 1) return '昨天'
  if (days < 7) return `${days}天前`
  if (days < 30) return `${Math.floor(days / 7)}周前`
  return target.toLocaleDateString('zh-CN')
}

const handleCardClick = () => emit('click', props.knowledge)
const handleFavorite = () => emit('favorite', props.knowledge)
const handleShare = () => emit('share', props.knowledge)
const handleTagClick = (tag: string) => emit('tagClick', tag)
const handleEdit = () => emit('edit', props.knowledge)
const handleDownload = () => emit('download', props.knowledge)
const handleReport = () => emit('report', props.knowledge)
</script>
```

```vue
        :message="estimateMessage"
        :type="estimateType"
        show-icon

<script setup lang="ts">
import { ref, computed, watch } from 'vue'
import {
  CheckCircleOutlined,
  RobotOutlined,
  UserOutlined
} from '@ant-design/icons-vue'

interface AuditorInfo {
  id: string
  name: string
  department: string
  workload: number
}

interface AuditConfig {
  mode: 'none' | 'auto' | 'manual'
  autoConfig?: any
  manualConfig?: any
}

const props = defineProps<{
  category?: string
  defaultMode?: 'none' | 'auto' | 'manual'
}>()

const emit = defineEmits<{
  change: [config: AuditConfig]
}>()

const auditMode = ref(props.defaultMode || 'manual')

const autoConfig = ref({
  model: 'standard',
  checkItems: ['quality', 'format', 'sensitive'],
  manualReview: true
})

const manualConfig = ref({
  auditor: '',
  priority: 'medium',
  note: ''
})

const auditorList = ref<AuditorInfo[]>([
  { id: '1', name: '王主管', department: '技术部', workload: 5 },
  { id: '2', name: '李专员', department: '产品部', workload: 3 },
  { id: '3', name: '张经理', department: '运营部', workload: 8 }
])

const estimateMessage = computed(() => {
  switch (auditMode.value) {
    case 'none':
      return '知识将直接发布，无需等待审核'
    case 'auto':
      return 'AI自动审核，预计处理时间：1-5分钟'
    case 'manual':
      const auditor = auditorList.value.find(a => a.id === manualConfig.value.auditor)
      if (auditor) {
        const estimateTime = auditor.workload < 5 ? '1个工作日' : '2-3个工作日'
        return `人工审核，预计处理时间：${estimateTime}`
      }
      return '请选择审核员'
    default:
      return ''
  }
})

const estimateType = computed(() => {
  switch (auditMode.value) {
    case 'none': return 'success'
    case 'auto': return 'info'
    case 'manual': return 'warning'
    default: return 'info'
  }
})

const handleModeChange = () => {
  emitConfig()
}

const emitConfig = () => {
  const config: AuditConfig = {
    mode: auditMode.value,
    autoConfig: auditMode.value === 'auto' ? autoConfig.value : undefined,
    manualConfig: auditMode.value === 'manual' ? manualConfig.value : undefined
  }
  emit('change', config)
}

watch([autoConfig, manualConfig], emitConfig, { deep: true })
</script>
```
